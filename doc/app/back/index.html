Backoffice authentication setup

Read credentials off a JSON file in this file's directory and initialize a Passport HTTP Basic strategy with those.

function readCredentials(cb) { fs.readFile(path.join(__dirname, 'credentials.json'), function(err, json) { if (err) console.warn("Missing backoffice credentials -> You won't be able to authenticate!"); var creds = JSON.parse(json || '{}'); if (creds.user && creds.password) console.log("Credentials loaded"); else console.log("One or more blank credential -> You won't be able to authenticate!"); cb(creds.user, creds.password); }); }
